014    package com.googlecode.jsendnsca;
015    
016    import java.io.IOException;
017    
018    /**
019     * Interface to be implemented by {@link PassiveCheckSender}'s
020     * 
021     * @author Raj.Patel
022     * @version 1.0
023     */
024    public interface PassiveCheckSender {
025        /**
026         * Send Passive Check
027         * 
028         * @param payload
029         *            the Passive Check message payload
030         * @throws NagiosException
031         *             thrown if an error occurs while sending the passive check
032         * @throws IOException
033         *             thrown if I/O error occurs while trying to establish
034         *             connection with nagios host
035         */
036        void send(MessagePayload payload) throws NagiosException, IOException;
037    }
