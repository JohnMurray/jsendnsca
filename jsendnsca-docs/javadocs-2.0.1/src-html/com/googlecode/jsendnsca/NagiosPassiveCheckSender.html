014    package com.googlecode.jsendnsca;
015    
016    import java.io.DataInputStream;
017    import java.io.IOException;
018    import java.io.OutputStream;
019    import java.net.InetSocketAddress;
020    import java.net.Socket;
021    import java.net.SocketException;
022    import java.net.SocketTimeoutException;
023    
024    import org.apache.commons.lang.Validate;
025    
026    import com.googlecode.jsendnsca.utils.IOUtils;
027    
028    /**
029     * This class is used to send a Passive Check to the Nagios NSCA add-on
030     *
031     * @author Raj.Patel
032     * @version 1.0
033     */
034    public class NagiosPassiveCheckSender implements PassiveCheckSender {
035    
036        private static final int INITIALISATION_VECTOR_SIZE = 128;
037    
038        private final NagiosSettings nagiosSettings;
039    
040        /**
041         * Construct a new {@link NagiosPassiveCheckSender} with the provided
042         * {@link NagiosSettings}
043         *
044         * @param nagiosSettings
045         *            the {@link NagiosSettings} to use to send the Passive Check
046         */
047        public NagiosPassiveCheckSender(NagiosSettings nagiosSettings) {
048            Validate.notNull(nagiosSettings, "nagiosSettings cannot be null");
049            this.nagiosSettings = nagiosSettings;
050        }
051    
052        /*
053         * (non-Javadoc)
054         *
055         * @see
056         * com.googlecode.jsendnsca.sender.INagiosPassiveCheckSender#send(com.googlecode
057         * .jsendnsca.sender.MessagePayload)
058         */
059        public void send(MessagePayload payload) throws NagiosException, IOException {
060            Validate.notNull(payload, "payload cannot be null");
061    
062            final Socket socket = new Socket();
063            final InetSocketAddress nagiosEndpoint = new InetSocketAddress(nagiosSettings.getNagiosHost(), nagiosSettings.getPort());
064    
065            OutputStream outputStream = null;
066            DataInputStream inputStream = null;
067            try {
068                connectWithTimeout(socket, nagiosEndpoint);
069                outputStream = socket.getOutputStream();
070                inputStream = new DataInputStream(socket.getInputStream());
071            } catch (IOException ioe) {
072                throw ioe;
073            }
074    
075            final byte[] initVector = readInitializationVector(inputStream);
076            int timeStamp = inputStream.readInt();
077    
078            try {
079    
080                final byte[] passiveCheckBytes = new PassiveCheckBytesBuilder()
081                    .withTimeStamp(timeStamp)
082                    .withLevel(payload.getLevel())
083                    .withHostname(payload.getHostname())
084                    .withServiceName(payload.getServiceName())
085                    .withMessage(payload.getMessage())
086                    .writeCRC()
087                    .encrypt(initVector,nagiosSettings)
088                    .toByteArray();
089    
090                outputStream.write(passiveCheckBytes);
091                outputStream.flush();
092            } catch (SocketTimeoutException ste) {
093                throw ste;
094            } catch (Exception e) {
095                throw new NagiosException("Error occurred while sending passive alert", e);
096            } finally {
097                IOUtils.closeQuietly(outputStream);
098                IOUtils.closeQuietly(inputStream);
099                IOUtils.closeQuietly(socket);
100            }
101        }
102    
103        private void connectWithTimeout(final Socket socket, final InetSocketAddress nagiosEndpoint) throws IOException, SocketException {
104            socket.connect(nagiosEndpoint, nagiosSettings.getConnectTimeout());
105            socket.setSoTimeout(nagiosSettings.getTimeout());
106        }
107    
108        private byte[] readInitializationVector(DataInputStream inputStream) throws NagiosException, SocketTimeoutException {
109            final byte[] initVector = new byte[INITIALISATION_VECTOR_SIZE];
110            try {
111                inputStream.readFully(initVector, 0, INITIALISATION_VECTOR_SIZE);
112                return initVector;
113            } catch (SocketTimeoutException ste) {
114                throw ste;
115            } catch (Exception e) {
116                throw new NagiosException("Can't read initialisation vector", e);
117            }
118        }
119    }
